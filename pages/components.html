// disabled 상태일 때는 변경 없음

var radios = document.querySelectorAll('.s1-radio');

radios.forEach(function(radio) {
  radio.addEventListener('click', function() {
    if (radio.classList.contains('is-disabled')) return;
    // 그룹 내 기존 선택 해제
    radios.forEach(function(r) { r.classList.remove('is-selected'); });
    // 클릭한 항목 선택
    radio.classList.add('is-selected');
  });
});

    // 3. 요일 헤더 렌더링
    function renderWeekdays() {
      weekdays.innerHTML = '';
      WEEK_LABELS.forEach(function(label, i) {
        var span = document.createElement('span');
        span.className = 's1-date-picker__weekday'
          + (i === 0 ? ' is-sunday' : '')
          + (i === 6 ? ' is-saturday' : '');
        span.textContent = label;
        weekdays.appendChild(span);
      });
    }

    // 4. 날짜 그리드 렌더링
    function renderGrid() {
      if (monthLabel) monthLabel.textContent =
        curYear + '.' + String(curMonth + 1).padStart(2, '0');
      grid.innerHTML = '';
      var firstDow  = new Date(curYear, curMonth, 1).getDay();
      var daysInMonth = new Date(curYear, curMonth + 1, 0).getDate();
      // 이전 달 trailing cells
      var prevDays = new Date(curYear, curMonth, 0).getDate();
      for (var i = 0; i < firstDow; i++)
        grid.appendChild(makeDay(new Date(curYear, curMonth - 1, prevDays - firstDow + i + 1), true));
      // 현재 달
      for (var d = 1; d <= daysInMonth; d++)
        grid.appendChild(makeDay(new Date(curYear, curMonth, d), false));
      // 다음 달 leading cells (6행 고정)
      var trailing = (firstDow + daysInMonth) % 7;
      if (trailing) for (var k = 1; k <= 7 - trailing; k++)
        grid.appendChild(makeDay(new Date(curYear, curMonth + 1, k), true));
    }

    // 5. 날짜 셀 생성
    function makeDay(date, isOther) {
      var dateKey = date.getFullYear() + '-'
        + String(date.getMonth() + 1).padStart(2, '0') + '-'
        + String(date.getDate()).padStart(2, '0');
      var isDisabled = !!disabledSet[dateKey]; // HD-6: 서비스 주입 날짜
      var btn = document.createElement('button');
      btn.type = 'button';
      btn.className = 's1-date-picker__day';
      var inner = document.createElement('span');
      inner.className = 'day-inner';
      inner.textContent = date.getDate();
      btn.appendChild(inner);
      if (isOther)    btn.classList.add('is-other-month'); // HD-7: 클릭 허용
      if (isSame(date, today))        btn.classList.add('is-today');
      if (isSame(date, selectedDate)) btn.classList.add('is-selected');
      if (isDisabled) { btn.disabled = true; btn.setAttribute('aria-disabled', 'true'); }
      if (!isDisabled) btn.addEventListener('click', function() {
        if (isOther) { curYear = date.getFullYear(); curMonth = date.getMonth(); }
        selectedDate = date;
        input.value = formatDate(date); // "YY.MM.DD"
        renderGrid(); closePanel();
      });
      return btn;
    }

    // 6. 이벤트 바인딩
    input.addEventListener('click', function(e) {
      e.stopPropagation();
      panel.hidden ? openPanel() : closePanel();
    });
    if (toggle) toggle.addEventListener('click', function(e) {
      e.stopPropagation();
      panel.hidden ? openPanel() : closePanel();
    });
    if (prevBtn) prevBtn.addEventListener('click', function() {
      if (--curMonth < 0) { curMonth = 11; curYear--; } renderGrid();
    });
    if (nextBtn) nextBtn.addEventListener('click', function() {
      if (++curMonth > 11) { curMonth = 0; curYear++; } renderGrid();
    });
    // ESC 닫기
    panel.addEventListener('keydown', function(e) {
      if (e.key === 'Escape') { closePanel(); input.focus(); }
    });
    // 외부 클릭 닫기
    document.addEventListener('click', function(e) {
      if (!picker.contains(e.target)) closePanel();
    });

    // 7. 유틸
    function isSame(a, b) {
      return a && b && a.getFullYear() === b.getFullYear()
        && a.getMonth() === b.getMonth()
        && a.getDate()  === b.getDate();
    }
    function formatDate(d) {
      var yy = String(d.getFullYear()).slice(-2);
      var mm = String(d.getMonth() + 1).padStart(2, '0');
      var dd = String(d.getDate()).padStart(2, '0');
      return yy + '.' + mm + '.' + dd;
    }
  });

// 고정 블록: currentPage가 속한 블록의 start/end/blockIndex 반환
function calcBlock(cp, tp) {
  const blockIndex = Math.floor((cp - 1) / BLOCK);
  const start = blockIndex * BLOCK + 1;
  const end   = Math.min(tp, start + BLOCK - 1);
  return { start, end, blockIndex };
}

// 페이지 번호 버튼 동적 렌더링 (블록 고정)
function renderNumbers() {
  const container = document.getElementById('pg-numbers');
  container.innerHTML = '';
  const tp = totalPages();
  const { start, end } = calcBlock(currentPage, tp);
  for (let p = start; p <= end; p++) {
    const btn = document.createElement('button');
    btn.className = 's1-pagination-number' + (p === currentPage ? ' is-selected' : '');
