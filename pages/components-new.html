// ── Checkbox ──
// label 클릭 → is-checked 토글
// disabled 상태일 때는 변경 없음

var checkbox = document.querySelector('.s1-checkbox');

checkbox.addEventListener('click', function() {
  if (checkbox.classList.contains('is-disabled')) return;
  checkbox.classList.toggle('is-checked');
});

// Indeterminate (테이블 전체선택 중간값)
// checkbox.classList.add('is-indeterminate');

// 그룹 내 복수 체크박스
var checkboxes = document.querySelectorAll('.s1-checkbox');
checkboxes.forEach(function(chk) {
  chk.addEventListener('click', function() {
    if (chk.classList.contains('is-disabled')) return;
    chk.classList.toggle('is-checked');
  });
});
